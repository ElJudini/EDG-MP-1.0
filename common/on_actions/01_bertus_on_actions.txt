#When adding effects directly in on_actions always consider if it would be best to do so through a scripted effect. on_actions with a lot of script in them are harder to get an overview of.

# province
on_startup = {
	if = {
		limit = {
			NOT = {has_country_flag = dev_clicks_count_var_flag}
			ai = no
		}
		set_country_flag = dev_clicks_count_var_flag
		set_variable = {
			which = dev_clicks_count_var
			value = 0
		}
	}
	if = {
		limit = {
			ai = no
		}
		clr_country_flag = disables_estate_church_flag
		clr_country_flag = disables_estate_nobles_flag
		clr_country_flag = disables_estate_burghers_flag
		clr_country_flag = disables_estate_dhimmi_flag
		clr_country_flag = disables_estate_brahmins_flag
		clr_country_flag = disables_estate_vaisyas_flag
		clr_country_flag = disables_estate_jains_flag
		clr_country_flag = disables_estate_eunuchs_flag
		clr_country_flag = disables_estate_cossacks_flag
		clr_country_flag = disables_estate_nomadic_tribes_flag
		clr_country_flag = disables_estate_rajput_flag
		clr_country_flag = disables_estate_maratha_flag
		clr_country_flag = disables_estate_janissaries_flag
		clr_country_flag = disables_estate_qizilbash_flag
		clr_country_flag = disables_estate_ghulams_flag
	}
}
on_adm_development = {
	owner = {
		if = {
			limit = {ai = no }
			change_variable = {which = dev_clicks_count_var value = 1}
		}
	}
}
